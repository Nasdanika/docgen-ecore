001package org.nasdanika.docgen.ecore.preferences;
002
003import org.eclipse.core.runtime.preferences.AbstractPreferenceInitializer;
004import org.eclipse.jface.preference.IPreferenceStore;
005
006import org.nasdanika.docgen.ecore.Activator;
007
008/**
009 * Class used to initialize default preference values.
010 */
011public class PreferenceInitializer extends AbstractPreferenceInitializer {
012
013        /*
014         * (non-Javadoc)
015         * 
016         * @see org.eclipse.core.runtime.preferences.AbstractPreferenceInitializer#initializeDefaultPreferences()
017         */
018        public void initializeDefaultPreferences() {
019                IPreferenceStore store = Activator.getDefault().getPreferenceStore();
020                store.setDefault(PreferenceConstants.P_BOOLEAN, true);
021                store.setDefault(PreferenceConstants.P_CHOICE, "choice2");
022                store.setDefault(PreferenceConstants.P_STRING,
023                                "Default value");
024        }
025
026}
